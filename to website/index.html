""" Reading input easily
Example: 
>> a,b,c,d = read_input((int,int,float,str))
1 2 3 hi
>> print('a:',a,' b:',b,' c:',c,' d:',d)
a: 1  b: 2  c: 3.0  d: hi
"""
def read_input(formatter):
    raw_in = input().split()
    assert len(formatter) == len(raw_in),'Formatter and input need to be of same length'
    return [format(raw) for format, raw in zip(formatter,raw_in)]
